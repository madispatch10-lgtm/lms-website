<script type="module">

import { getFirestore, doc, setDoc } from "https://www.gstatic.com/firebasejs/12.12.1/firebase-firestore.js";

  authDomain: "lms-training-7bb26.firebaseapp.com",
  projectId: "lms-training-7bb26",
  storageBucket: "lms-training-7bb26.firebasestorage.app",
  messagingSenderId: "791631109001",
  appId: "1:791631109001:web:f5d7fc5c2755ae01649f58",
  measurementId: "G-XKKPYPXYSP"
};

const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);

let attemptsLeft = 3;
let currentUser = null;
let currentQuestions = [];

const salesQuiz = [
{q:"What is cold calling?", a:"calling"},
{q:"What is lead generation?", a:"lead"},
{q:"What is closing?", a:"closing"}
];

const dispatchQuiz = [
{q:"What is load booking?", a:"load"},
{q:"What is broker?", a:"broker"},
{q:"What is rate?", a:"rate"}
];

window.register = async function(){
const email = document.getElementById('email').value;
const password = document.getElementById('password').value;
try {
await createUserWithEmailAndPassword(auth, email, password);
alert("Registered");
} catch(e){ alert(e.message); }
};

window.login = async function(){
const email = document.getElementById('email').value;
const password = document.getElementById('password').value;
try {
const res = await signInWithEmailAndPassword(auth, email, password);
currentUser = res.user;
document.getElementById('loginBox').classList.add('hidden');
document.getElementById('dashboardBox').classList.remove('hidden');
} catch(e){ alert(e.message); }
};

window.logout = async function(){
await signOut(auth);
location.reload();
};

window.startQuiz = function(){
attemptsLeft = 3;
const track = document.getElementById('track').value;

document.getElementById('dashboardBox').classList.add('hidden');
document.getElementById('quizBox').classList.remove('hidden');

currentQuestions = track === 'sales' ? salesQuiz : dispatchQuiz;

let html = "";
currentQuestions.forEach((q,i)=>{
html += `<div class='question'>${i+1}. ${q.q}<br><input id='q${i}'></div>`;
});

document.getElementById('questions').innerHTML = html;
document.getElementById('attempts').innerText = "Attempts: " + attemptsLeft;
};

window.submitQuiz = async function(){
let correct = true;

currentQuestions.forEach((q,i)=>{
const ans = document.getElementById('q'+i).value.toLowerCase();
if(!ans.includes(q.a)) correct = false;
});

if(correct){
const track = document.getElementById('track').value;

if(currentUser){
await setDoc(doc(db, "users", currentUser.uid), {
course: track,
completed: true
});
}

document.getElementById('quizBox').classList.add('hidden');
document.getElementById('courseBox').classList.remove('hidden');

document.getElementById('courseContent').innerHTML =
track === 'sales'
? '<li>Cold calling</li><li>Closing</li>'
: '<li>Load booking</li><li>Negotiation</li>';

} else {
attemptsLeft--;
if(attemptsLeft <= 0){
alert("Failed");
location.reload();
} else {
document.getElementById('attempts').innerText = "Wrong! Attempts: " + attemptsLeft;
}
}
};
